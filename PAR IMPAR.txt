Algoritmo "par impar"

Var
n,i,x:inteiro

Inicio
escreva("Quantos numeros voce vai digitar ?:")
leia(n)

para i de 1 ate n faca
  escreva("Digite um numero:")
  leia(x)
  
  se x=0 entao
    escreval("NULO")
  senao
    se x % 2 = 0 então
     se x < 0 entao
       escreval("Par Negativo")
     senao
       escreval("Par Positivo")
     fimse
    senao
      se x < 0 entao
        escreval("Impar Negativo")
      senao
        escreval("Impar Positivo")
      fimse
    fimse
  fimse
  
fimpara
