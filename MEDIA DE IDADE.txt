Algoritmo "Media de Idade"

Var

idade,cont,soma:inteiro
media:real

Inicio

soma <- 0
cont <- 0
escreval("Digite as idades: ")
leia(idade)

enquanto idade >= 0 faca

   soma <- soma + idade
   cont <- cont + 1

   leia(idade)
   
fimenquanto

se cont = 0 então
  escreva("IMPOSSIVEL CALCULAR")
senao
  media <- soma/cont
  escreva("Media = ",media:8:2)

fimse


Fimalgoritmo
